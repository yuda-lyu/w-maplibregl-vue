            let p = this.panelCompassRose
            if (isestr(p.iconSrc)) return p.iconSrc
            if (p.withPanel) return p.iconSrcDark
            return this.baseMapColorShade === 'dark' ? p.iconSrcLight : p.iconSrcDark

            let shade = ''
            each(get(this, 'panelBaseMaps.baseMaps', []), (v) => {
                if (v.visible && isestr(v.colorShade)) {
                    shade = v.colorShade; return false
                }

        /** 比例尺文字 */

            let lat = this.currentCenter[0] || 23.5
            let mpp = 156543.03392 * Math.cos(lat * Math.PI / 180) / Math.pow(2, this.currentZoom)
            let d = mpp * 80
            return d >= 1000 ? Math.round(d / 1000) + ' km' : Math.round(d) + ' m'
        },

            return {
                display: 'inline-block',
                transform: `perspective(160px) rotateX(${tilt}deg)`,
                transition: 'transform 0.08s linear',
                transformOrigin: 'center center',

            if (!isnum(zoom)) zoom = 6 //MapLibre GL因圖磚以512px為基準, zoom會差1

                    version: 8,

                center: [center[1], center[0]],
                zoom,
                attributionControl: false,
                antialias: true, //讓 3D 地形邊緣更平滑（須付出輕微效能成本，但 3D 模式必要）
            })

            vo.map.on('load', () => {
                vo.mapLoaded = true
                vo.applyProjection()
                vo.applyBaseMaps()
                vo.applyTerrain()
                // 確保所有資料在 map load 時已處理（debounce 可能尚未觸發）
                vo.ensureDataProcessed()
                vo.applyAllDataLayers()
            })
            vo.map.on('mousemove', (e) => {
                vo.showLoc = { lat: dig(e.lngLat.lat, 7), lng: dig(e.lngLat.lng, 7) }
            })
            vo.map.on('zoomend', () => {
                // let z = Math.round(vo.map.getZoom()); vo.opt.zoom = z; vo.currentZoom = z
                let z = (vo.map.getZoom()); vo.opt.zoom = z; vo.currentZoom = z
                // auto projection：zoom 跨過閾值時自動切換
                if (vo.projection === '') vo.applyProjection()
            })

                let c = vo.map.getCenter(); let nc = [c.lat, c.lng]
                if (!isEqual(vo.opt.center, nc)) vo.opt.center = nc
                vo.currentCenter = nc

        // ===== 底圖 =====

        applyBaseMaps() {
            let vo = this
            if (!vo.map || !vo.mapLoaded) return
            let baseMaps = get(vo, 'panelBaseMaps.baseMaps', [])
            let addBaseMapLayer = (bm, k) => {
                let srcId = `basemap-src-${k}`; let layerId = `basemap-layer-${k}`
                if (vo.map.getLayer(layerId)) vo.map.removeLayer(layerId)
                if (vo.map.getSource(srcId)) vo.map.removeSource(srcId)
                if (bm.type === 'vector') {
                    let u = bm.url; if (u.startsWith('//')) u = 'https:' + u
                    vo.map.addSource(srcId, { type: 'vector', url: u })
                    let layerType = bm.layerType || 'fill'
                    let sourceLayer = bm.layer || ''
                    let op = bm.opacity != null ? bm.opacity : 0.8
                    let paint = {}
                    if (layerType === 'fill-extrusion') {
                        paint = {
                            'fill-extrusion-color': ['coalesce', ['get', 'colour'], '#aaaaaa'],
                            'fill-extrusion-height': ['interpolate', ['linear'], ['zoom'], 15, 0, 15.05, ['coalesce', ['get', 'render_height'], ['get', 'height'], 0]],
                            'fill-extrusion-base': ['interpolate', ['linear'], ['zoom'], 15, 0, 15.05, ['coalesce', ['get', 'render_min_height'], ['get', 'min_height'], 0]],
                            'fill-extrusion-opacity': op,
                        }
                    }
                    else if (layerType === 'fill') {
                        paint = { 'fill-color': '#aaaaaa', 'fill-opacity': op }
                    }
                    vo.map.addLayer({ 'id': layerId, 'type': layerType, 'source': srcId, 'source-layer': sourceLayer, paint, 'layout': { visibility: bm.visible ? 'visible' : 'none' } })
                    return
                }
                let tiles = []
                if (bm.type === 'wms') {
                    let u = bm.url; if (!u.includes('?')) u += '?'
                    u += `SERVICE=WMS&VERSION=1.1.1&REQUEST=GetMap&LAYERS=${bm.layers || ''}&STYLES=&FORMAT=image/png&TRANSPARENT=true&SRS=EPSG:3857&WIDTH=256&HEIGHT=256&BBOX={bbox-epsg-3857}`
                    tiles = [u]
                }
                else {
                    let u = bm.url
                    tiles = u.includes('{s}') ? ['a', 'b', 'c'].map((s) => u.replace('{s}', s)) : [u]
                    tiles = tiles.map((t) => t.startsWith('//') ? 'https:' + t : t)
                }
                let op = bm.opacity != null ? bm.opacity : 1
                vo.map.addSource(srcId, { type: 'raster', tiles, tileSize: 256 })
                vo.map.addLayer({ id: layerId, type: 'raster', source: srcId, paint: { 'raster-opacity': op }, layout: { visibility: bm.visible ? 'visible' : 'none' } })
            }
            // 第一輪：底圖（colorShade 非空），確保先渲染在下方
            each(baseMaps, (bm, k) => {
                if (isestr(bm.colorShade)) addBaseMapLayer(bm, k)
            })
            // 第二輪：疊加圖層（colorShade 空字串），後渲染在底圖上方
            each(baseMaps, (bm, k) => {
                if (!isestr(bm.colorShade)) addBaseMapLayer(bm, k)
            })
        },

        applyTerrain() {
            let vo = this
            if (!vo.map || !vo.mapLoaded) return

            // 清除舊設定：先釋放 terrain，再移除 layer/source
            try {
                vo.map.setTerrain(null)
            }
            catch (e) {}
            if (vo.map.getLayer('terrain-hillshade')) vo.map.removeLayer('terrain-hillshade')
            if (vo.map.getSource('terrain-hillshade-src')) vo.map.removeSource('terrain-hillshade-src')
            if (vo.map.getSource('terrain-src')) vo.map.removeSource('terrain-src')

            let tm = vo.terrainMap
            if (!isobj(tm)) return
            let tsrc = get(tm, 'terrainSource', null)
            if (!isobj(tsrc) || !isestr(tsrc.url)) return

            // 添加 raster-dem terrain source
            let tu = tsrc.url; if (tu.startsWith('//')) tu = 'https:' + tu
            let terrainSrcSpec = {
                type: 'raster-dem',
                tiles: [tu],
                tileSize: tsrc.tileSize || 256,
                encoding: tsrc.encoding || 'terrarium',
            }
            if (isnum(tsrc.maxzoom)) terrainSrcSpec.maxzoom = parseInt(tsrc.maxzoom)
            try {
                vo.map.addSource('terrain-src', terrainSrcSpec)
            }
            catch (e) {
                console.error('[WMaplibreglVue] applyTerrain addSource error:', e)
                return
            }
            // 啟用 3D 地形高程
            try {
                vo.map.setTerrain({ source: 'terrain-src', exaggeration: isnum(tm.exaggeration) ? cdbl(tm.exaggeration) : 1.5 })
                // console.log('[WMaplibreglVue] setTerrain OK, tiles:', tu)
            }
            catch (e) {
                console.error('[WMaplibreglVue] applyTerrain setTerrain error:', e)
            }

            // 添加 hillshade（山坡陰影）
            let hsrc = get(tm, 'hillshadeSource', null)
            if (isobj(hsrc) && isestr(hsrc.url)) {
                let hu = hsrc.url; if (hu.startsWith('//')) hu = 'https:' + hu
                // 若 hillshade URL 與 terrain URL 相同，共用 terrain-src 避免重複載入
                let hillshadeSourceId = 'terrain-src'
                if (hu !== tu) {
                    let hillshadeSrcSpec = {
                        type: 'raster-dem',
                        tiles: [hu],
                        tileSize: hsrc.tileSize || 256,
                        encoding: hsrc.encoding || 'terrarium',
                    }
                    if (isnum(hsrc.maxzoom)) hillshadeSrcSpec.maxzoom = parseInt(hsrc.maxzoom)
                    vo.map.addSource('terrain-hillshade-src', hillshadeSrcSpec)
                    hillshadeSourceId = 'terrain-hillshade-src'
                }
                // 插入在第一個 data layer 之前，確保 hillshade 不覆蓋圖資
                let insertBefore = null
                for (let lid of vo.trackedLayerIds) {
                    if (vo.map.getLayer(lid)) {
                        insertBefore = lid; break
                    }
                }
                let hillshadeSpec = {
                    id: 'terrain-hillshade',
                    type: 'hillshade',
                    source: hillshadeSourceId,
                    paint: {
                        'hillshade-shadow-color': '#473B24',
                        'hillshade-exaggeration': isnum(tm.hillshadeExaggeration) ? cdbl(tm.hillshadeExaggeration) : 0.5,
                    },
                }
                if (insertBefore && vo.map.getLayer(insertBefore)) {
                    vo.map.addLayer(hillshadeSpec, insertBefore)
                }
                else {
                    vo.map.addLayer(hillshadeSpec)
                }
            }
        },

                let sz = get(ps, 'size', 10); let fl = get(ps, 'fillColor', 'rgba(0,150,255,0.65)'); let lc = get(ps, 'lineColor', 'rgba(255,255,255,1)'); let lw = get(ps, 'lineWidth', 1)
                ps.points = map(ps.points, (pt, kpt) => {

                    if (!isestr(get(pt, 'title', null))) pt.title = ''; if (!isestr(get(pt, 'msg', null))) pt.msg = ''
                    return { id: `pointSet-${kps}-point-${kpt}`, ...pt, radius: get(pt, 'size', null) || sz, fillColor: get(pt, 'fillColor', null) || fl, lineColor: get(pt, 'lineColor', null) || lc, lineWidth: get(pt, 'lineWidth', null) || lw, funSetsClick: fc }
                })
                return { id: `pointSet-${kps}`, ...ps, funSetsClick: fc }
            })
            vo.changeItemsDebounce('changePointSets')
            // 方案五：若 style 尚未載入完成，等待 idle 事件後再渲染（避免 renderPointSets 被跳過且無重試）
            if (vo.map) {
                if (vo.mapLoaded) {
                    vo.renderPointSets()
                }
                else {
                    vo.map.once('idle', () => {
                        vo.renderPointSets()
                    })
                }
            }
        },
        /**
         * 建立方向性彈窗（支援 top/bottom/left/right + 自動翻轉）
         * 渲染後會以實際 DOM 尺寸做二次校正，若彈窗超出圖台邊界則翻轉至對向。
         * @param {Array} lngLat - [lng, lat]
         * @param {HTMLElement} content - DOM 內容
         * @param {String} position - 'top'|'bottom'|'left'|'right'
         * @param {Number} gap - 與錨點的間距（px）
         * @param {Object} extraOpts - 其他 Popup 選項
         * @param {Array} [anchorOffset=[0,0]] - 額外偏移 [dx, dy]（正x=右移, 正y=下移，px）
         * @returns {maplibregl.Popup}
         */
        createDirectionalPopup(lngLat, content, position, gap, extraOpts, anchorOffset) {
            let vo = this; if (!vo.map) return null
            gap = isNumber(gap) ? Math.abs(gap) : 5
            // anchorOffset（正x=右, 正y=下）
            let aoX = 0; let aoY = 0
            if (isarr(anchorOffset) && anchorOffset.length >= 2) {
                aoX = anchorOffset[0] || 0
                aoY = anchorOffset[1] || 0
            }
            let posMap = {
                'top': { anchor: 'bottom', offset: [aoX, -gap + aoY] },
                'bottom': { anchor: 'top', offset: [aoX, gap + aoY] },
                'left': { anchor: 'right', offset: [-gap + aoX, aoY] },
                'right': { anchor: 'left', offset: [gap + aoX, aoY] },
            }
            let flipMap = { 'top': 'bottom', 'bottom': 'top', 'left': 'right', 'right': 'left' }

            // 判斷在指定方向上是否超出邊界
            let checkOverflow = (dir, projPt, cw, ch, elW, elH) => {
                if (dir === 'top') return projPt.y - gap - elH < 0
                if (dir === 'bottom') return projPt.y + gap + elH > ch
                if (dir === 'left') return projPt.x - gap - elW < 0
                if (dir === 'right') return projPt.x + gap + elW > cw
                return false
            }

            // 第一階段：以預估尺寸決定初始方向
            let chosenPos = position
            try {
                let proj = vo.map.project(lngLat)
                let ct = vo.map.getContainer()
                let cw = ct.clientWidth; let ch = ct.clientHeight
                let estW = 220; let estH = 160
                if (checkOverflow(position, proj, cw, ch, estW, estH)) {
                    chosenPos = flipMap[position] || position
                }
            }
            catch (e) { /* 投影失敗時使用原始方向 */ }

            let pos = posMap[chosenPos] || posMap['top']
            let popup = new maplibregl.Popup({ ...extraOpts, anchor: pos.anchor, offset: pos.offset })
                .setLngLat(lngLat).setDOMContent(content).addTo(vo.map)

            // 第二階段：渲染後以實際 DOM 尺寸做二次校正
            try {
                let popupEl = popup.getElement()
                if (popupEl) {
                    let rect = popupEl.getBoundingClientRect()
                    let realW = rect.width; let realH = rect.height
                    let proj = vo.map.project(lngLat)
                    let ct = vo.map.getContainer()
                    let cw = ct.clientWidth; let ch = ct.clientHeight
                    // 若當前方向以實際尺寸檢查仍溢出，且對向不會溢出，則翻轉
                    if (checkOverflow(chosenPos, proj, cw, ch, realW, realH)) {
                        let flipped = flipMap[chosenPos] || chosenPos
                        if (!checkOverflow(flipped, proj, cw, ch, realW, realH)) {
                            chosenPos = flipped
                            let newPos = posMap[flipped]
                            popup.remove()
                            popup = new maplibregl.Popup({ ...extraOpts, anchor: newPos.anchor, offset: newPos.offset })
                                .setLngLat(lngLat).setDOMContent(content).addTo(vo.map)
                        }
                    }
                }
            }
            catch (e) { /* 二次校正失敗時維持當前方向 */ }

            // 儲存建立參數供拖曳/縮放時動態翻轉偵測使用
            popup._dirMeta = {
                lngLat,
                position, // 原始期望方向
                chosenDir: chosenPos, // 實際使用方向
                gap,
                extraOpts,
                anchorOffset: anchorOffset || null,
                contentHtml: content.innerHTML, // 保存內容 HTML 以供重建
            }

            return popup
        },
        /** 將 base64/URL 圖片載入並以指定大小註冊到地圖 sprite atlas */
        registerIconImage(key, src, tw, th) {
            let vo = this
            if (!vo.map || vo.map.hasImage(key)) return Promise.resolve()
            return new Promise((resolve) => {
                let img = new Image()
                img.onload = () => {
                    if (!vo.map || vo.map.hasImage(key)) {
                        resolve(); return
                    }
                    let c = document.createElement('canvas'); c.width = tw; c.height = th
                    let ctx = c.getContext('2d'); ctx.drawImage(img, 0, 0, tw, th)
                    let d = ctx.getImageData(0, 0, tw, th)
                    vo.map.addImage(key, { width: tw, height: th, data: new Uint8Array(d.data.buffer) })
                    resolve()
                }
                img.onerror = () => resolve()
                img.src = src
            })
        },

        /**
         * 渲染 pointSets（canvas 模式：circle layer + symbol layer）
         * 每個 pointSet 建立一個 GeoJSON source + circle/symbol 兩個 layer
         */
        renderPointSets() {
            let vo = this; if (!vo.map) return
            // 過渡期清理：移除舊的 DOM markers
            vo.clearTrackedMarkersByPrefix('point')

            each(vo.pointSets, (ps, kps) => {
                let srcId = `point-${kps}-src`
                let circleId = `point-${kps}-circle`
                let symbolId = `point-${kps}-symbol`
                let clusterCircleId = `point-${kps}-cluster-circle`
                let clusterCountId = `point-${kps}-cluster-count`

                if (!ps.visible) {
                    // 不可見：移除此 pointSet 的所有 layer/source（含叢集 layers）
                    each([circleId, symbolId, clusterCircleId, clusterCountId], (lid) => {
                        if (vo.map.getLayer(lid)) vo.map.removeLayer(lid)
                    })
                    if (vo.map.getSource(srcId)) vo.map.removeSource(srcId)
                    vo.trackedLayerIds = vo.trackedLayerIds.filter(x => x !== circleId && x !== symbolId && x !== clusterCircleId && x !== clusterCountId)
                    vo.trackedSourceIds = vo.trackedSourceIds.filter(x => x !== srcId)
                    return
                }

                // pointSet 層級設定
                let psType = get(ps, 'type', 'circle')
                let psIconSrc = get(ps, 'iconSrc', '')
                let psIconSize = get(ps, 'iconSize', [25, 41])
                let psIconAnchor = get(ps, 'iconAnchor', null)
                // let psPAnchor = get(ps, 'popupAnchor', [0, 0])
                // let psTAnchor = get(ps, 'tooltipAnchor', [0, 0])
                let psIconKey = `point-icon-ps-${kps}`

                let features = []
                // 方案六：重置計數器（每次 renderPointSets 整體重建時重置）
                if (kps === 0) vo._featureIdCounter = 0
                let iconLoadTasks = []
                let psIconRegistered = false
                let iconOffsets = {} // 收集各 point 的 icon offset，key 為 _kpt

                each(ps.points, (pt, kpt) => {
                    let ll = get(pt, 'latLng', null); if (!isarr(ll) || size(ll) < 2) return
                    let ptype = get(pt, 'type', null) || psType
                    let props = { _kps: kps, _kpt: kpt, _ptype: ptype, _id: pt.id || '' }

                    if (ptype === 'circle') {
                        props._radius = pt.radius || get(ps, 'size', 10)
                        props._fillColor = pt.fillColor || get(ps, 'fillColor', 'rgba(0,150,255,0.65)')
                        props._lineColor = pt.lineColor || get(ps, 'lineColor', 'rgba(255,255,255,1)')
                        props._lineWidth = pt.lineWidth || get(ps, 'lineWidth', 1)
                    }
                    else if (ptype === 'icon') {
                        let ptIconSrc = get(pt, 'iconSrc', null) || psIconSrc
                        let ptIconSize = get(pt, 'iconSize', null) || psIconSize
                        let ptIconAnchor = get(pt, 'iconAnchor', null) || psIconAnchor

                        // 檢查是否為有效 icon 設定（與 WMaplibreglVue 的 getIconParam 一致）
                        let eff = isestr(ptIconSrc) && isarr(ptIconSize) && size(ptIconSize) === 2 && isarr(ptIconAnchor) && size(ptIconAnchor) === 2
                        if (!eff) {
                            // 無有效 icon 設定，使用預設 icon（google map 預設 marker）
                            ptIconSrc = uiRes.iconPoint
                            ptIconSize = [24, 40]
                            ptIconAnchor = [ptIconSize[0] / 2, ptIconSize[1]] // 底部中心
                            // 同步設定預設 popupAnchor / tooltipAnchor（若外部未提供）
                            if (!isarr(get(pt, 'popupAnchor', null)) && !isarr(get(ps, 'popupAnchor', null))) {
                                pt._defPopupAnchor = [0, -ptIconSize[1] / 1.0]
                            }
                            if (!isarr(get(pt, 'tooltipAnchor', null)) && !isarr(get(ps, 'tooltipAnchor', null))) {
                                pt._defTooltipAnchor = [0, -ptIconSize[1] / 1.0]
                            }
                        }

                        let ptIw = ptIconSize[0]
                        let ptIh = ptIconSize[1]

                        // 註冊 icon 圖片
                        let iconKey
                        if (eff && isestr(get(pt, 'iconSrc', null))) {
                            // point 級別自訂 icon
                            iconKey = `point-icon-pt-${kps}-${kpt}`
                            if (!vo.map.hasImage(iconKey)) iconLoadTasks.push(vo.registerIconImage(iconKey, ptIconSrc, ptIw, ptIh))
                        }
                        else if (eff && isestr(psIconSrc)) {
                            // pointSet 級別自訂 icon
                            if (!psIconRegistered) {
                                let w = isarr(psIconSize) ? psIconSize[0] : 25; let h = isarr(psIconSize) ? psIconSize[1] : 41
                                if (!vo.map.hasImage(psIconKey)) iconLoadTasks.push(vo.registerIconImage(psIconKey, psIconSrc, w, h))
                                psIconRegistered = true
                            }
                            iconKey = psIconKey
                        }
                        else {
                            // 預設 icon
                            let defKey = `point-icon-default`
                            if (!vo.map.hasImage(defKey)) iconLoadTasks.push(vo.registerIconImage(defKey, ptIconSrc, ptIw, ptIh))
                            iconKey = defKey
                        }
                        props._iconKey = iconKey

                        // iconAnchor 語義：
                        // 圖片上第 [x, y] 像素位置對齊地圖座標，由圖片左上角起算
                        // MapLibre icon-anchor:'center' 時，icon-offset 是相對圖片中心的偏移
                        // 偏移量 = 圖片中心 - iconAnchor（正值=右/下偏移）
                        let centerX = ptIw / 2; let centerY = ptIh / 2
                        let offX = centerX - ptIconAnchor[0]
                        let offY = centerY - ptIconAnchor[1]
                        iconOffsets[kpt] = [offX, offY]
                    }
                    // 方案六：為每個 Feature 指派唯一數值 id，加速 MapLibre 內部索引與 queryRenderedFeatures
                    let featureId = vo._featureIdCounter++
                    features.push({ type: 'Feature', id: featureId, geometry: { type: 'Point', coordinates: [ll[1], ll[0]] }, properties: props })
                })

                let geojsonData = { type: 'FeatureCollection', features }

                // 建構 icon-offset expression：使用 match 依各 point 的 _kpt 查表取得對應 offset
                let iconOffsetExpr
                let offKeys = Object.keys(iconOffsets)
                if (offKeys.length === 0) {
                    // 無 icon point
                    iconOffsetExpr = ['literal', [0, 0]]
                }
                else {
                    // 檢查所有 offset 是否都相同
                    let firstOff = iconOffsets[offKeys[0]]
                    let allSame = offKeys.every((k) => iconOffsets[k][0] === firstOff[0] && iconOffsets[k][1] === firstOff[1])
                    if (allSame) {
                        // 所有 icon 偏移相同，使用固定值
                        iconOffsetExpr = ['literal', firstOff]
                    }
                    else {
                        // 不同 point 有不同偏移，使用 match expression
                        let matchArgs = ['match', ['get', '_kpt']]
                        each(offKeys, (k) => {
                            matchArgs.push(parseInt(k, 10))
                            matchArgs.push(['literal', iconOffsets[k]])
                        })
                        matchArgs.push(['literal', firstOff]) // fallback
                        iconOffsetExpr = matchArgs
                    }
                }

                let doRender = () => {
                    if (!vo.map) return
                    if (vo.map.getSource(srcId)) {
                        // 已存在：用 setData 更新，不閃爍
                        vo.map.getSource(srcId).setData(geojsonData)
                        // 方案五：強制重設 symbol layer 的 icon-image 與 icon-offset，確保 icon 圖片註冊後能被正確引用
                        if (vo.map.getLayer(symbolId)) {
                            vo.map.setLayoutProperty(symbolId, 'icon-image', ['get', '_iconKey'])
                            vo.map.setLayoutProperty(symbolId, 'icon-offset', iconOffsetExpr)
                        }
                    }
                    else {
                        // 新建 source + layer
                        // 方案六：啟用 generateId 作為備援，確保 Feature 有唯一 id
                        let srcOpts = { type: 'geojson', data: geojsonData, generateId: true }
                        // 叢集化：在 source 層級啟用 MapLibre 原生 cluster 支援（內部使用 supercluster）
                        if (vo.clusterPoints) {
                            srcOpts.cluster = true
                            srcOpts.clusterRadius = vo.clusterPointsRadius
                            srcOpts.clusterMaxZoom = vo.clusterPointsMaxZoom
                        }
                        vo.map.addSource(srcId, srcOpts)
                        // circle layer（叢集 feature 不含 _ptype，自然被 filter 排除）
                        vo.map.addLayer({
                            id: circleId,
                            type: 'circle',
                            source: srcId,
                            filter: ['==', ['get', '_ptype'], 'circle'],
                            paint: {
                                'circle-radius': ['get', '_radius'],
                                'circle-color': ['get', '_fillColor'],
                                'circle-stroke-color': ['get', '_lineColor'],
                                'circle-stroke-width': ['get', '_lineWidth'],
                            }
                        })
                        // symbol layer：使用 icon-anchor:'center'，透過 match expression 取各 point 的 offset
                        vo.map.addLayer({
                            id: symbolId,
                            type: 'symbol',
                            source: srcId,
                            filter: ['==', ['get', '_ptype'], 'icon'],
                            layout: {
                                'icon-image': ['get', '_iconKey'],
                                'icon-size': 1,
                                'icon-anchor': 'center',
                                'icon-offset': iconOffsetExpr,
                                'icon-allow-overlap': true,
                                'icon-ignore-placement': true,
                            }
                        })
                        vo.trackedSourceIds.push(srcId)
                        vo.trackedLayerIds.push(circleId, symbolId)

                        // 叢集化相關 layers
                        if (vo.clusterPoints) {
                            // 建構動態 step expression：依 clusterPointsLevel* 參數產生
                            let buildStepExpr = (vals) => {
                                if (vo.clusterPointsLevelNum <= 1) return vals[0]
                                let expr = ['step', ['get', 'point_count'], vals[0]]
                                for (let li = 0; li < vo.clusterPointsLevelValues.length; li++) {
                                    expr.push(vo.clusterPointsLevelValues[li], vals[li + 1])
                                }
                                return expr
                            }
                            let colorExpr = buildStepExpr(vo.clusterPointsLevelFillColors)
                            let radiusExpr = buildStepExpr(vo.clusterPointsLevelRadius)
                            let strokeColorExpr = buildStepExpr(vo.clusterPointsLevelLineColors)
                            let strokeWidthExpr = buildStepExpr(vo.clusterPointsLevelLineWidths)
                            let textSizeExpr = buildStepExpr(vo.clusterPointsLevelTextSizes)
                            let textColorExpr = buildStepExpr(vo.clusterPointsLevelTextColors)
                            // 叢集圓圈 layer：依 point_count 動態調整大小與顏色
                            vo.map.addLayer({
                                id: clusterCircleId,
                                type: 'circle',
                                source: srcId,
                                filter: ['has', 'point_count'],
                                paint: {
                                    'circle-color': colorExpr,
                                    'circle-radius': radiusExpr,
                                    'circle-stroke-width': strokeWidthExpr,
                                    'circle-stroke-color': strokeColorExpr,
                                }
                            })
                            // 叢集計數文字 layer
                            vo.map.addLayer({
                                id: clusterCountId,
                                type: 'symbol',
                                source: srcId,
                                filter: ['has', 'point_count'],
                                layout: {
                                    'text-field': '{point_count_abbreviated}',
                                    'text-size': textSizeExpr,
                                },
                                paint: {
                                    'text-color': textColorExpr,
                                }
                            })
                            vo.trackedLayerIds.push(clusterCircleId, clusterCountId)

                            // 點擊叢集圓圈時放大展開
                            vo.map.on('click', clusterCircleId, (e) => {
                                let features = vo.map.queryRenderedFeatures(e.point, { layers: [clusterCircleId] })
                                if (!features.length) return
                                let clusterId = features[0].properties.cluster_id
                                let src = vo.map.getSource(srcId)
                                if (!src) return
                                let result = src.getClusterExpansionZoom(clusterId)
                                if (result && typeof result.then === 'function') {
                                    result.then((zoom) => {
                                        vo.map.easeTo({ center: features[0].geometry.coordinates, zoom })
                                    }).catch(() => {})
                                }
                            })
                            vo.map.on('mouseenter', clusterCircleId, () => {
                                vo.map.getCanvas().style.cursor = 'pointer'
                            })
                            vo.map.on('mouseleave', clusterCircleId, () => {
                                vo.map.getCanvas().style.cursor = ''
                            })
                        }

                        // 事件處理函數
                        let handlePointEvent = (e, action) => {
                            if (!e.features || !e.features.length) return
                            let f = e.features[0]; let p = f.properties
                            let ptData = get(vo, `pointSets.${p._kps}.points.${p._kpt}`, null)
                            let psData = get(vo, `pointSets.${p._kps}`, null)
                            if (!ptData || !psData) return
                            // 使用 feature 座標（point 中心）而非滑鼠位置，確保 offset 定位正確
                            let coords = f.geometry.coordinates
                            action(e, ptData, psData, p, coords)
                        }
                        let onPointClick = (e) => {
                            handlePointEvent(e, (ev, ptData, psData, p, coords) => {
                                if (isfun(ptData.funSetsClick)) ptData.funSetsClick({ ev, point: ptData, pointSet: psData, pointSets: vo.pointSets })
                                // popupAnchor（point 級別 → pointSet 級別 → _def → 預設值）
                                let popAnchor = get(ptData, 'popupAnchor', null) || get(psData, 'popupAnchor', null)
                                if (!isarr(popAnchor) || popAnchor.length < 2) {
                                    if (p._ptype === 'circle') {
                                        let r = p._radius || get(psData, 'size', 10)
                                        popAnchor = [0, -(r / 1.0)]
                                    }
                                    else {
                                        // 優先使用 renderPointSets 中設定的預設值
                                        popAnchor = get(ptData, '_defPopupAnchor', null)
                                        if (!isarr(popAnchor)) {
                                            let iSz = get(ptData, 'iconSize', null) || get(psData, 'iconSize', null) || [25, 41]
                                            popAnchor = [0, -(iSz[1] / 1.0)]
                                        }
                                    }
                                }
                                vo.activePoint = ptData; vo.activePointSet = psData
                                vo.$nextTick(() => {
                                    let refEl = vo.$refs.refPointPopup; if (!refEl) return
                                    let html = refEl.innerHTML.trim(); if (!html) return
                                    vo.closeAllPopupsIfOnlyone()
                                    let ct = document.createElement('div'); ct.innerHTML = html
                                    vo.featurePopup = vo.createDirectionalPopup(coords, ct, vo.popupPosition, 0, { maxWidth: 'none' }, popAnchor)
                                    vo.featurePopupOwner = `pointSets.${p._kps}`
                                })
                            })
                        }
                        let onPointEnter = (e) => {
                            vo.map.getCanvas().style.cursor = 'pointer'
                            handlePointEvent(e, (ev, ptData, psData, p, coords) => {
                                // tooltipAnchor（point 級別 → pointSet 級別 → _def → 預設值）
                                let tipAnchor = get(ptData, 'tooltipAnchor', null) || get(psData, 'tooltipAnchor', null)
                                if (!isarr(tipAnchor) || tipAnchor.length < 2) {
                                    if (p._ptype === 'circle') {
                                        tipAnchor = [0, 0]
                                    }
                                    else {
                                        // 優先使用 renderPointSets 中設定的預設值
                                        tipAnchor = get(ptData, '_defTooltipAnchor', null)
                                        if (!isarr(tipAnchor)) {
                                            let iSz = get(ptData, 'iconSize', null) || get(psData, 'iconSize', null) || [25, 41]
                                            tipAnchor = [0, -(iSz[1] / 1.0)]
                                        }
                                    }
                                }
                                vo.activePoint = ptData; vo.activePointSet = psData
                                vo.$nextTick(() => {
                                    let refEl = vo.$refs.refPointTooltip; if (!refEl) return
                                    let html = refEl.innerHTML.trim(); if (!html) return
                                    vo.hideFeatureTooltip()
                                    let ct = document.createElement('div'); ct.innerHTML = html
                                    vo.featureTooltip = vo.createDirectionalPopup(coords, ct, vo.tooltipPosition, 0, { closeButton: false, closeOnClick: false, maxWidth: 'none', className: 'wlv2-tooltip' }, tipAnchor)
                                    vo.featureTooltipOwner = `pointSets.${p._kps}`
                                })
                            })
                        }
                        let onPointLeave = () => {
                            vo.map.getCanvas().style.cursor = ''; vo.hideFeatureTooltip()
                        }

                        vo.map.on('click', circleId, onPointClick)
                        vo.map.on('click', symbolId, onPointClick)
                        vo.map.on('mouseenter', circleId, onPointEnter)
                        vo.map.on('mouseenter', symbolId, onPointEnter)
                        vo.map.on('mouseleave', circleId, onPointLeave)
                        vo.map.on('mouseleave', symbolId, onPointLeave)
                    }
                }

                if (iconLoadTasks.length > 0) {
                    Promise.all(iconLoadTasks).then(doRender).catch(() => doRender())
                }
                else {
                    doRender()
                }
            })
        },

        changePolylineSets() {
            let vo = this; let sets = get(vo, 'opt.polylineSets', null); if (!isarr(sets))sets = []; sets = cloneDeep(sets)
            let eff = map(sets, (v) => omit(v, 'visible')); if (isEqual(vo.effPolylineSetsTemp, eff)) return; vo.effPolylineSetsTemp = eff

        /**
         * 點是否在環（ring）內部（Ray Casting 演算法）
         * @param {Array} pt - [lng, lat] 座標
         * @param {Array} ring - [[lng, lat], ...] 環座標陣列
         * @returns {Boolean}
         */
        pointInRing(pt, ring) {
            let inside = false
            for (let i = 0, j = ring.length - 1; i < ring.length; j = i++) {
                let xi = ring[i][0]; let yi = ring[i][1]
                let xj = ring[j][0]; let yj = ring[j][1]
                if (((yi > pt[1]) !== (yj > pt[1])) &&
                    (pt[0] < (xj - xi) * (pt[1] - yi) / (yj - yi) + xi)) {
                    inside = !inside
                }
            }
            return inside
        },

         */

            if (!vo.contourSubCounts) vo.contourSubCounts = {} // 追蹤每個 contourSet 的子層數量

            each(vo.contourSets, (cs, kcs) => {
                let prevCount = vo.contourSubCounts[kcs] || 0

                if (!cs.visible) {

                    for (let i = 0; i < prevCount; i++) {
                        let sid = `contour-${kcs}-src-${i}`

                        let lid = `contour-${kcs}-line-${i}`

                        if (vo.map.getSource(sid)) vo.map.removeSource(sid)

                        vo.trackedSourceIds = vo.trackedSourceIds.filter(x => x !== sid)
                    }
                    vo.contourSubCounts[kcs] = 0
                    return
                }

                // 取得 points
                let points = get(cs, 'points', [])
                if (!isearr(points)) return

                // 呼叫 calcContours 計算等值線
                let calcOpt = {
                    containInner: get(cs, 'polygonsContainInner', null),
                    clipInner: get(cs, 'polygonsClipInner', null),
                    clipOuter: get(cs, 'polygonClipOuter', null),
                    thresholds: get(cs, 'thresholds', null),
                }
                let polygonSets = calcContours(points, calcOpt)
                if (!isarr(polygonSets) || polygonSets.err) return

                // 為每個 polygonSet 計算色彩與樣式
                let n = polygonSets.length - 1
                polygonSets = map(polygonSets, (ps, k) => {
                    let color = vo.getContourColor(k, n, cs.gradient, cs.getColor, ps)
                    let lineColor = isestr(cs.lineColor) ? cs.lineColor : color
                    let fillColor = color
                    return { ...ps, color, lineColor, fillColor }
                })

                // 觸發 contourRefresh 以生成圖例
                vo.contourRefresh({ polygonSets }, cs, kcs)

                let newCount = polygonSets.length

                // 移除多餘的舊子層（若本次子層數 < 上次子層數）
                for (let i = newCount; i < prevCount; i++) {
                    let sid = `contour-${kcs}-src-${i}`
                    let fid = `contour-${kcs}-fill-${i}`
                    let lid = `contour-${kcs}-line-${i}`
                    if (vo.map.getLayer(fid)) vo.map.removeLayer(fid)
                    if (vo.map.getLayer(lid)) vo.map.removeLayer(lid)
                    if (vo.map.getSource(sid)) vo.map.removeSource(sid)
                    vo.trackedLayerIds = vo.trackedLayerIds.filter(x => x !== fid && x !== lid)
                    vo.trackedSourceIds = vo.trackedSourceIds.filter(x => x !== sid)
                }

                // 渲染每個 polygonSet（增量更新）
                each(polygonSets, (ps, kps) => {
                    let latLngs = get(ps, 'latLngs', [])
                    if (!isearr(latLngs)) return

                    let sid = `contour-${kcs}-src-${kps}`
                    let fid = `contour-${kcs}-fill-${kps}`
                    let lid = `contour-${kcs}-line-${kps}`

                    // latLngs 是 MultiPolygon 格式（緯經度），轉為 GeoJSON 格式（經緯度）
                    let coordinates = map(latLngs, (polygon) => {
                        return map(polygon, (ring) => {
                            if (isarr(ring) && isarr(ring[0])) {
                                return map(ring, (pt) => [pt[1], pt[0]])
                            }
                            return [ring[1], ring[0]]
                        })
                    })

                    // 方案六：為等值線 Feature 加上唯一數值 id
                    let featureId = vo._featureIdCounter++
                    let geojsonData = {
                        type: 'Feature',
                        id: featureId,
                        geometry: {
                            type: 'MultiPolygon',
                            coordinates,
                        }
                    }

                    if (vo.map.getSource(sid)) {
                        // 已存在：用 setData 更新，不閃爍
                        vo.map.getSource(sid).setData(geojsonData)
                    }
                    else {
                        // 不存在：新建 source + layer + 綁定事件
                        vo.map.addSource(sid, { type: 'geojson', data: geojsonData, generateId: true })
                        vo.map.addLayer({
                            id: fid,
                            type: 'fill',
                            source: sid,
                            paint: { 'fill-color': ps.fillColor, 'fill-opacity': cs.fillOpacity }
                        })
                        vo.map.addLayer({
                            id: lid,
                            type: 'line',
                            source: sid,
                            paint: { 'line-color': ps.lineColor, 'line-width': cs.lineWidth }
                        })
                        vo.trackedSourceIds.push(sid)
                        vo.trackedLayerIds.push(fid, lid)

                        // click 事件
                        vo.map.on('click', fid, (e) => {
                            vo.clickContour(e, ps, kps, cs, kcs, vo.contourSets)
                        })
                        // mouseenter
                        vo.map.on('mouseenter', fid, (e) => {
                            vo.map.getCanvas().style.cursor = 'pointer'
                            if (cs.changeStyleWhenHover) {
                                let lineColorH = isestr(cs.lineColorHover) ? cs.lineColorHover : ps.color
                                vo.map.setPaintProperty(fid, 'fill-opacity', cs.fillOpacityHover)
                                vo.map.setPaintProperty(lid, 'line-color', lineColorH)
                                vo.map.setPaintProperty(lid, 'line-width', cs.lineWidthHover)
                            }
                            // 顯示 tooltip slot
                            vo.activeContourSet = cs
                            vo.$nextTick(() => {
                                let el = vo.$refs.refContourTooltip
                                if (!el) return
                                let html = el.innerHTML.trim()
                                if (!html) return
                                vo.hideFeatureTooltip()
                                let container = document.createElement('div')
                                container.innerHTML = html
                                vo.featureTooltip = vo.createDirectionalPopup(e.lngLat, container, vo.tooltipPosition, 10, { closeButton: false, closeOnClick: false, maxWidth: 'none', className: 'wlv2-tooltip' })
                                vo.featureTooltipOwner = `contourSets.${kcs}`
                            })
                        })
                        // mouseleave
                        vo.map.on('mouseleave', fid, () => {
                            vo.map.getCanvas().style.cursor = ''
                            if (cs.changeStyleWhenHover) {
                                vo.map.setPaintProperty(fid, 'fill-opacity', cs.fillOpacity)
                                vo.map.setPaintProperty(lid, 'line-color', ps.lineColor)
                                vo.map.setPaintProperty(lid, 'line-width', cs.lineWidth)
                            }
                            vo.hideFeatureTooltip()
                        })
                    }
                })

                // 更新子層數量
                vo.contourSubCounts[kcs] = newCount
            })
        },

        /** 等值線圖例生成（依 WMaplibreglVue.vue 的 contourRefresh 邏輯） */
        contourRefresh(data, contourSet, kcontourSet) {
            let legends = map(data.polygonSets, 'range')

            function getText(k, range) {
                let low = range.low
                let up = range.up
                let delimiter = '-'
                if (isNumber(contourSet.legendNumDig)) {
                    low = dig(range.low, contourSet.legendNumDig)
                    up = dig(range.up, contourSet.legendNumDig)
                }
                if (isfun(contourSet.legendTextFormater)) {
                    let r = contourSet.legendTextFormater({ low, up, legends, index: k })
                    low = get(r, 'low')
                    up = get(r, 'up')
                    delimiter = get(r, 'delimiter', delimiter)
                }
                return { low, up, delimiter }
            }

            let legend = map(data.polygonSets, (v, k) => {
                let textExt = ''
                if (isfun(contourSet.legendTextExtra)) {
                    textExt = contourSet.legendTextExtra({ k, n: size(data.polygonSets), polygonSet: v })
                }
                return {
                    ...getText(k, v.range),
                    textExt,
                    color: v.color,
                    arrow: false,
                    index: k,
                }
            })

            legend = reverse(legend)
            contourSet.legend = legend
        },

        /**
         * 平移至指定座標
         * @param {Array} latLng - [lat, lng]
         * @param {Object} [opt] - { ratioHorizontal, ratioVertical, funLatLng }
         */
        panTo(latLng, opt = {}) {
            let vo = this
            if (!isearr(latLng) || !vo.map) return

            // ratioHorizontal：左側起算之水平縮小比率
            let ratioHorizontal = get(opt, 'ratioHorizontal')
            if (isnum(ratioHorizontal)) {
                ratioHorizontal = cdbl(ratioHorizontal)
            }

            // ratioVertical：上方起算之垂直縮小比率
            let ratioVertical = get(opt, 'ratioVertical')
            if (isnum(ratioVertical)) {
                ratioVertical = cdbl(ratioVertical)
            }

            // funLatLng：自訂座標轉換函式
            let funLatLng = get(opt, 'funLatLng')

            // 取得目前地圖邊界
            let bds = vo.map.getBounds()
            let ne = bds.getNorthEast()
            let sw = bds.getSouthWest()
            let lngRange = ne.lng - sw.lng
            let latRange = ne.lat - sw.lat

            // 反算以外部 latLng 為中心之角點座標
            let lngMin = latLng[1] - lngRange / 2
            let lngMax = latLng[1] + lngRange / 2
            let latMin = latLng[0] - latRange / 2
            let latMax = latLng[0] + latRange / 2

            // 計算偏移後的新中心
            let lngNew = null
            if (isNumber(ratioHorizontal)) {
                let lngMaxNew = lngMax - lngRange * ratioHorizontal
                lngNew = (lngMin + lngMaxNew) / 2
            }
            let latNew = null
            if (isNumber(ratioVertical)) {
                let latMinNew = latMin + latRange * ratioVertical
                latNew = (latMinNew + latMax) / 2
            }

            // 組合新座標
            let latLngNew = cloneDeep(latLng)
            if (isNumber(ratioVertical)) {
                latLngNew[0] = latNew
            }
            if (isNumber(ratioHorizontal)) {
                latLngNew[1] = lngNew
            }

            // 執行自訂轉換函式
            if (isfun(funLatLng)) {
                latLngNew = funLatLng(latLng, {
                    bds,
                    lngRange,
                    latRange,
                    lngMin,
                    lngMax,
                    latMin,
                    latMax,
                    ratioHorizontal,
                    ratioVertical,
                    latLngNew,
                })
            }

            // 執行平移 [lat,lng] → [lng,lat]
            vo.map.panTo([latLngNew[1], latLngNew[0]])
        },
        centerTo(ll) {
            this.panTo(ll)

        /**
         * 針對單一 popup/tooltip 做方向偵測與翻轉
         * @param {String} propName - 'featurePopup' 或 'featureTooltip'
         */
        _recheckSinglePopupDir(propName) {
            let vo = this
            let popup = vo[propName]
            if (!popup || !popup._dirMeta) return

            let meta = popup._dirMeta
            let { lngLat, position, chosenDir, gap, extraOpts, anchorOffset } = meta

            // anchorOffset（正x=右, 正y=下），與 MapLibre offset 方向一致
            let aoX = 0; let aoY = 0
            if (isarr(anchorOffset) && anchorOffset.length >= 2) {
                aoX = anchorOffset[0] || 0
                aoY = anchorOffset[1] || 0
            }

            let posMap = {
                'top': { anchor: 'bottom', offset: [aoX, -gap + aoY] },
                'bottom': { anchor: 'top', offset: [aoX, gap + aoY] },
                'left': { anchor: 'right', offset: [-gap + aoX, aoY] },
                'right': { anchor: 'left', offset: [gap + aoX, aoY] },
            }
            let flipMap = { 'top': 'bottom', 'bottom': 'top', 'left': 'right', 'right': 'left' }

            try {
                let popupEl = popup.getElement()
                if (!popupEl) return
                let rect = popupEl.getBoundingClientRect()
                let elW = rect.width; let elH = rect.height
                let proj = vo.map.project(lngLat)
                let ct = vo.map.getContainer()
                let cw = ct.clientWidth; let ch = ct.clientHeight

                let checkOverflow = (dir) => {
                    if (dir === 'top') return proj.y - gap - elH < 0
                    if (dir === 'bottom') return proj.y + gap + elH > ch
                    if (dir === 'left') return proj.x - gap - elW < 0
                    if (dir === 'right') return proj.x + gap + elW > cw
                    return false
                }

                let overflow = checkOverflow(chosenDir)
                let newDir = chosenDir

                if (overflow) {
                    // 當前方向溢出，嘗試翻轉至對向
                    let flipped = flipMap[chosenDir]
                    if (!checkOverflow(flipped)) {
                        newDir = flipped
                    }
                }
                else if (chosenDir !== position) {
                    // 目前處於翻轉狀態，檢查原始方向是否已有足夠空間，若有則翻回
                    if (!checkOverflow(position)) {
                        newDir = position
                    }
                }

                // 方向無變化則不重建
                if (newDir === chosenDir) return

                // 需要翻轉：以保存的 contentHtml 重建 popup
                let newPos = posMap[newDir]
                let newContent = document.createElement('div')
                newContent.innerHTML = meta.contentHtml

                popup.remove()
                let newPopup = new maplibregl.Popup({ ...extraOpts, anchor: newPos.anchor, offset: newPos.offset })
                    .setLngLat(lngLat).setDOMContent(newContent).addTo(vo.map)

                // 繼承 meta 並更新方向
                newPopup._dirMeta = { ...meta, chosenDir: newDir, contentHtml: newContent.innerHTML }
                vo[propName] = newPopup
            }
            catch (e) { /* 偵測失敗時維持現狀 */ }
        },

         * @returns {Array|null} [lat, lng] 或 null
         */
        popupFeatureById(id) {
            let vo = this; if (!isestr(id) || !vo.map) return null
            // 解析 id 找到對應的數據集
            let center = null
            let featureData = null
            let ownerPath = ''
            let type = ''
            if (id.startsWith('polylineSet-')) {
                let idx = parseInt(id.replace('polylineSet-', ''), 10)
                let pls = get(vo, 'polylineSets.' + idx, null); if (!pls) return null
                featureData = pls
                ownerPath = `polylineSets.${idx}`
                type = 'polyline'
                // 取折線中點
                let lls = get(pls, 'latLngs', [])
                if (isearr(lls)) {
                    let seg = isarr(lls[0]) && isNumber(lls[0][0]) ? lls : lls[0]
                    if (isearr(seg)) {
                        let mid = seg[Math.floor(seg.length / 2)]; center = [mid[0], mid[1]]
                    }
                }
            }
            else if (id.startsWith('polygonSet-')) {
                let idx = parseInt(id.replace('polygonSet-', ''), 10)
                let pg = get(vo, 'polygonSets.' + idx, null); if (!pg) return null
                featureData = pg
                ownerPath = `polygonSets.${idx}`
                type = 'polygon'
                // 取多邊形第一環重心
                let lls = get(pg, 'latLngs', [])
                if (isearr(lls)) {
                    let ring = isarr(lls[0]) && isNumber(lls[0][0]) ? lls : lls[0]
                    if (isearr(ring)) {
                        let sumLat = 0; let sumLng = 0
                        each(ring, (ll) => {
                            sumLat += ll[0]; sumLng += ll[1]
                        })
                        center = [sumLat / ring.length, sumLng / ring.length]
                    }
                }
            }
            if (!center || !featureData) return null
            // 更新 active 資料（觸發 slot 響應式更新）
            if (type === 'polyline') vo.activePolylineSet = featureData
            else if (type === 'polygon') vo.activePolygonSet = featureData
            // 等 Vue 更新 DOM 後才取 slot 內容
            vo.$nextTick(() => {
                let refName = type === 'polyline' ? 'refPolylinePopup' : 'refPolygonPopup'
                let el = vo.$refs[refName]
                let html = el ? el.innerHTML.trim() : ''
                // 若無 slot 內容，則使用預設 HTML
                if (!html) {
                    html = `<div style="padding:8px"><b>${featureData.title || ''}</b><br/>${featureData.msg || ''}</div>`
                }
                vo.closeAllPopupsIfOnlyone()
                let ct = document.createElement('div')
                ct.innerHTML = html
                vo.featurePopup = vo.createDirectionalPopup([center[1], center[0]], ct, vo.popupPosition, 5, { maxWidth: 'none' })
                vo.featurePopupOwner = ownerPath
            })
            return center
        },

        modifyItemsVisible(fun) {
            let vo = this; if (!isfun(fun)) return
            let hiddenKeys = []
            each(vo.items, (v, k) => {
                let b = fun(v, k)
                if (!b && v.visible) hiddenKeys.push(v.updatePath.replace('.visible', ''))
                v.visible = b; set(vo, v.updatePath, b); set(vo.opt, v.updatePath, b)
            })
            // 只關閉屬於被隱藏 set 的 popup/tooltip，不影響其他 set 已開啟的 popup
            each(hiddenKeys, (setKey) => {
                vo.closePopupByOwner(setKey)
            })
            if (vo.map && vo.mapLoaded)vo.applyAllDataLayers()
        },
    },
}
